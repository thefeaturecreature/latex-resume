% Load unified bullet store and query helpers.
% Requires uselua.tex to be loaded first (data.lua provides company/skill metadata).
\begin{luacode}
  local bulletsPath = kpse.find_file('../data/bullets.lua')
  if bulletsPath then
    local ok, err = pcall(dofile, bulletsPath)
    if not ok then tex.print('ERROR loading bullets.lua: ' .. tostring(err)) end
  else
    tex.print('ERROR: bullets.lua not found')
  end

  local queriesPath = kpse.find_file('../includes/queries.lua')
  if queriesPath then
    local ok, err = pcall(dofile, queriesPath)
    if not ok then tex.print('ERROR loading queries.lua: ' .. tostring(err)) end
  else
    tex.print('ERROR: queries.lua not found')
  end
\end{luacode}

% Proficiency block backed by getItemsForSkill().
% Metadata (name, description) still sourced from data.lua proficiencies table.
% Usage: \profblockQ{SkillKey}{1,2,3}        — with description
%        \profblockQ*{SkillKey}{1,2,3}       — without description
\NewDocumentCommand{\profblockQ}{s m m}{%
  \proficiency{\directlua{tex.print(proficiencies["#2"].name)}}%
  \IfBooleanF{#1}{\profdesc{\directlua{tex.print(proficiencies["#2"].description)}}}%
  \bullets{\directlua{renderItems(getItemsForSkill("#2"), "#3")}}%
}

% Experience block backed by getBulletsForCompany().
% Metadata (title, dates, company, location, desc, body) still sourced from data.lua experiences table.
% Optional arg: comma-separated bullet indices to render. Omit to show no bullets (proficiency-resume default).
% Usage: \expblockQ[1,2,3]{companykey}
\newcommand{\expblockQ}[2][]{%
  \jobtitle{\directlua{tex.print(experiences["#2"].title)}}%
  \dates{\directlua{tex.print(experiences["#2"].dateStart)}}{\directlua{tex.print(experiences["#2"].dateEnd)}}%
  \company{\directlua{tex.print(experiences["#2"].company)}} \location{\directlua{tex.print(experiences["#2"].location)}} \desc{\directlua{tex.print(experiences["#2"].desc)}}%
  \body{\directlua{tex.print(experiences["#2"].body)}}%
  \ifx&#1&\else\bullets{\directlua{renderBullets(getBulletsForCompany("#2"), "#1")}}\fi%
}
