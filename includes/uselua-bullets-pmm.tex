% PMM bullet store loader and block commands.
% Requires uselua.tex first (loads experiences + proficiencies, defines renderBullets/renderItems).
% Also requires uselua-bullets.tex first (loads bullets.lua + queries.lua).
\begin{luacode}
  local pmmProfPath = kpse.find_file('../includes/pmm/proficiencies.lua')
  if pmmProfPath then
    local ok, err = pcall(dofile, pmmProfPath)
    if not ok then tex.print('ERROR loading pmm/proficiencies.lua: ' .. tostring(err)) end
  else
    tex.print('ERROR: pmm/proficiencies.lua not found')
  end
\end{luacode}

% PMM proficiency block.
% Pulls name/description from pmm_proficiencies, bullets via getPMMItemsForSkill().
% Usage: \profblockPMMQ{SkillKey}{1,2,3}
\newcommand{\profblockPMMQ}[2]{%
  \proficiency{\directlua{tex.print(pmm_proficiencies["#1"].name)}}%
  \profdesc{\directlua{tex.print(pmm_proficiencies["#1"].description)}}%
  \bullets{\directlua{renderItems(getPMMItemsForSkill("#1"), "#2")}}%
}

% PMM experience block.
% Renders title bolded, followed by (pmm_title) in normal weight if pmm_title is set.
% Uses pmm_body if present, falls back to body.
% Usage: \expblockPMMQ[1,2,3]{companykey}
\newcommand{\expblockPMMQ}[2][]{%
  \jobtitle{\directlua{tex.print(experiences["#2"].title)}}%
  \directlua{
    local e = experiences["#2"]
    if e.pmm_title then
      tex.print("\\ (\\mdseries " .. e.pmm_title .. ")")
    end
  }%
  \dates{\directlua{tex.print(experiences["#2"].dateStart)}}{\directlua{tex.print(experiences["#2"].dateEnd)}}%
  \company{\directlua{tex.print(experiences["#2"].company)}} \location{\directlua{tex.print(experiences["#2"].location)}} \desc{\directlua{tex.print(experiences["#2"].desc)}}%
  \body{\directlua{
    local e = experiences["#2"]
    tex.print(e.pmm_body or e.body)
  }}%
  \ifx&#1&\else\bullets{\directlua{renderBullets(getPMMBulletsForCompany("#2"), "#1")}}\fi%
}
